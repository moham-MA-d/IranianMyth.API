میگه ناچار ناگذیر هم در منوی ناچاره میگه ناچار به ناخواست یک پیش آمده مهم و خطیر روخ داده پس این جلسه رو رایزن به همراه نویسنده یعنی اینا که مشورت میدادن نویسنده شاه رو هم آوردن یک نامه به نویسه یک نامه فرمود پس شهریار کیکابوزشا نوشتن برای رستم نامدار یه نامه نوشتن برای رستم گیوورده رو ببره به رستم بده نخست آفرین کرد بر پهلوان که بیدار دل باش و روشن روان میدونیم هم که رستم از کیکابوزشا خوشش نمیاد چون معتقده که کیکابوزشا بیخرده و کم هم درد سر درست نکرده تا الان جنگ مازندران و بعد هاماوران و بعد رفتن به آسمون و عرض سبارم داستان چون دوبارشو اسیر شد یه بارشو سقوط کرد داشت میمور رستم باید میرفت جمعش میکرد بنابراین نامه رو با ستایش پهلوان جهان آغاز میکنه من بگه امیدوارم که همیشه روشن روان حوشیار و بیدار باشی چون این بادکندر جهان جز تو کس نباشد به هر کار فریاد از کجایی بیا من بازگیر کردم همواره و همیشه چونان بوده که در جهان و هر کار مهم و هر پیش آمده ناگواری که برای ایران زمین پیش میاد کسی به جز تو یاریگر ایران نخواهد بود کسی از پسش بر نخواهد اومد یه نامه از گجده هم به دستم رسید که خوندم شرایط وخیمه چو برخان نمان نامه گجده هم نشستند گردان به پیشم به هم به نزد تو آردگران مای گیو بر این گونه دیدند گردان نیو نامه گجده هم رو خوندم سپه سالاران در کنار من همه با هم رای زنی کردیم و خب سپه سالاران پهلوان سلاح رو در این دونستند که گیو گران مایه نامه گجده هم رو به نزد تو برسونه و نتیجه این ماجرا این شده که ما تصمیم گرفتیم که تو بیایی به داده ایران برسی چجوری هم بیایی باز این خیلی مهمه چون نامه به خانی به روز و به شب هر موقع که رسید بهت روز بود خوندی شب بود خوندی مکن داستان را گشاده دولب مگر با سواران بسیار حوش ززابل برانی براری خروش این دولب گشاده کردن در معنی لب باز کردن ممکنه 

معنی اولی که به ذهنتون میرسه این باشه که این نامه رو خوندی لب رو باز نکن به کسی چیزی نکن ولی یه معنی دقیق تری هم داره یعنی به اندازه باز کردن لب از لب هم معتل نکن راه بیفت چون نامه به خانی به روز و به شب مکن داستان را گشاده دولب مگر با سواران بسیار حوش ززابل برانی براری خروش تنها کاری که میکنی باید این باشه که با سپایان و حوشیار در زابل بتازی بانگ و هنگامه جنگ رو بپا کنی بیای اوضاع خرابه بدان سان که گشده هم از او یاد کرد جز از تو نباشد و را هم نبرد من کسی رو ندارم با این به جنگه اینجوری که گشده هم اون رو توصیف کرده به نظر نمیاد کسی به غیر از تو حریفش بشه 

نامه رو نوشت قرارم از گیو ببره به گیوانگهی گفت برسان دود انان تقاور به باید بسود بسودن رو که در قسمت قبل رو جبه صحبت کردیم پسودن این کلمه رو خیلی استفاده میکنه در شاهنامه این لمس کردن ساییدن دست بردن انان تقاور بسودن یعنی افسار در اصف رو در دست گرفتن کنایه از پیوسته در حال حرکت بودن افسار اصف رو رحانه کنی بگیوانگهی گفت برسان دود انان تقاور به باید بسود یا بسود بهش گفت مثل دود باید بر اصفت سوار بشی افسار اصفم میگیری دستت ویل نمی کنی تا برسی نباید چون نزدیک رستم شوی به زابل بمانی و گربغ نوی رسیدی ورشتار بیارش سبر نمی کنی اونجاها نباید به زابل بمونی و استراحت کنی رسیدی دستشو بگی ورداری بیار اگر شب رسی روز را بازگرد بگویش که تنگ اندر آمد نبرد شب رسیدی روز بر میگردی و بهش میگی که جنگ به زودی در خواهد گرفت اینا رسیدن به دج سپید یا بعد از اون میخواند بیان به سمت سرزمین ایران بنابراین جنگ به زودی در خواهد گرفت وقت صرف نمی کنی یا جم کن بیا 

گیو هم چی کار کنه از اون آمه بستد به کردار آب برفت و نجستیچ آرام و خواب آب اینجا تو شاهنامه اصلا از آب برای ساختن تصویرهای مختلفی با وجه شبه گوناگون از آب استفاده شده اینجا وجه شبه سرعت و روانی و شتاب آبه بنابراین میگه از اون نامه بستد به کردار آب یعنی گیو نامه رو از کابوزشاه گرفت و مثل آب تند و چابک رفت راه افتاد برفت و نجستیچ آرام و خواب آرام و خواب رو کنار گذاشت مستقیم رفت برسه به رستم حالا نزدیک رستم هم میرسیم تو نزدیکی زابلستان رسید خروش تلایه به دستان رسید تلایه یعنی دیدبان دستان هم که نام زال پدر رستم پسر سام بود گیو به نزدیک های زابلستان که رسید زال بانگ دیدبان رو از خبر اومدن گیو داشت حشدار میداد شنید و خب رستم هم باید پذیراش بشه دیگه تهمتن پذیره شدش با سپا نهادند بر سر بزرگان کلا مهمه بلاخره یک فرستاده ای از سمت شاهنشاه اومده رستم و زال در حال فرمان روایی بر سیستان هستند خب بیشتر الان قدرت دست رستمه زال پدرش در کنارشه به عنوان نماد خرد و یک نامه و یک فرستاده ای از برحال شاهنشاه اومده دیگر چقدر این شاهنشاه تحتیل باشه برحال از نهاد پادشاهی پیام اومده یک زوابطی با رایتش تهمتن پذیره شدش با سپا نهادند بر سر بزرگان کلا این کلا بر سر گذاشتند دیگه باید بدونید سرکردگان و بزرگان شهر وقتی میخواستند برند پیشواز کسی به نشانه احترام اساسا ملبس به یک لباس رسمی میشدند بزرگان شهر کلاهای نشاندار خودشون رو رو سر میذاشتند و خب آمه مردم لباس متفاوتی داشتند به خصوص در مورد پوشش سر یعنی کلاهداران یک طبقه ممتاز بین درباریان و فرماندهان لشگر بودند که اینا کلاه به سر میذاشتند کلاسه روستن با سپاهیانش سپاهیان و سران سیستان به نشانه احترام به این فرستاده از سمت پادشاه کلاه روی سر گذاشتند لباس رسمی به تنشون کردند و رفتند به پیشواز گیو 

پیاده شدش گیو و گردان به هم هران کس که برزین بود از بیش و کم برزین بودن یعنی برزین نشسته و سوار بود از اسپندر آمد گوه نامدار از ایران بپرسید و از شهریان وقتی نزدیک هم رسیدن گیو و دلاوران سیستان هر کسی که سواره بود خواه بلند مرتبه بود خواه کوتح مرتبه و کوتح رتبه همگی در برابر رستم از اسپاشون پیاده شدن خب گوه نامدار رستمم اومد از اسپندر آمد گوه نامدار از ایران بپرسید و از شهریار رستم از اسپش اومد پایین در باره اوزا ایران و پادشاه ایران یعنی که ای کابوسشاه از گیو پرسجو کرد 

و خب رفتند به سمت ایوان رستم زره سوی ایوان رستم شدن ببودن یک لخت و دم برزدند گیو و سپاهش یک راست به سمت خاخ رستم روانه شدن که اونجا نفس تازه کردن و از اینجا بر رفتار رستم خیلی جالبه به گفتانچ بشنید و نامه بداد زه سهراب چندی سخن کرد یاد بنابراین اگر تهمینه تا الان در دوازده سالگی به پسرش نگفته که پدرتو رستمه 

به پدرش یعنی رستم هم که هدایی فرستاده برای تهمینه در این مدت به او هم نگفته که نام فرزند رو سهراب گذاشته نگفته که یکی یل تنومندی شده و نبراین این دو همونه میشناسن پس اگر در نام نامی از سهراب میخونه او نمیدونه که نام فرزند خودش سهرابه چون ممکنه تهمینه نام دیگری رو بیان کرده باشه اصلا نامی به میان نیاورده باشه برای رستم رستم دیگه بعد از این دوازده سال گذاشته به سمنگان بر نگشته چون به نظر میاد که این هماغوشی پنهانی رو اختده و نام نشانی از فرزندش به غیر از اون بازوبندی که یک مهره نشان داشت نداره اون رو هم باید ویژوال ببینه با چشمش باید بسری ببینه که بفهمه این بچش 

گیف در باره این پیش آمد گفتگو کرد با رستم که آقا دیژ سپید و یه سهراب نامی با سپای تورانیان اومده تسخیر میخواد بکنه نگهبانان و جنگاوران دیژ سپید دیدن نمیتونن مقاممت کنن در رفتن نامه دادن به پادشاه پادشاه همین نامه رو داده به من همون نامه نامه که گشده هم نوشتر رو که ما داریم اینجا رو تخلیه میکنیم و من بیارم بدم به شما و شما رو بردارم ببرم آجام باید بریم کار جمع بکنیم 

واکنش تهمتن بسیار جالبه از اینجا به این عبیات دقیقت بکنید حالا ما داریم کم کم میریم به پیشواز رزم رستم و سهراب و اونجا با هم حرفها داریم برای زدن که خب در داستان اینگوران دیگونه است که نه سهراب رستم رو میشناسه و نه رستم سهراب رو ولی نشانه هایی هست که به نظر میرسه هم برای سهراب و هم برای رستم قابل تشخیصه این موضوع که اون پدر منه و این پسر منه ولی خب چی باعث میشه که این جنگ شکل بگیره تا انتها میپردازی میشه تهمتن چو بشنید و نامه بخاند بخندید از آن کار و خیره بماند که این دوباره داره بزرگ نمایی میکنه حرف بی خود بی جد داره میزنه این که ای که آبوز شها دوباره که ماننده سام گرد از مهان سواری پدید آمدن در جهان آره باشه دوتا پدید اومده در جهان میان بزرگان پهلوانی مثل سام سوار پدید اومده مثل با بزرگ من از کجا پدید اومده 

باید نصر نمو هم باشه دیگه غیر از این باشه که نداریم که سواریز ترکان میامد پدید که اینجوریه ترکان غلط کردن از آزادگانی نباشد شگفت سی ترکان چونین یاد نتفان گرفت چی میگی بزرگ نکنین بابا برخواستن سام دلاور از بین ایرانیان شگفت آور نیست داره از نجاد ما از پشت من حتی یک فرزندی اینچونین زاده بشه که در تن اومندی و اندازه سام دلاور باشه ولی همچین چیزی رو به ترانیان نسبت نده که از سپاه ترانیان یکی اومده یه همچین چیزی من دارم یک سام جدیدی تو اینا میبینم مثل اون میتونه به جنگه اون کاره نیستن اونو اگه باشه باید از نجاد من باشه من از دخت شاه سمنگان یکی پسر دارم و باشد اون کودکی نمیتونه تخمین بزنه این رو که این ممکنه بچه خودش باشه ولی نشانه هایی هست الان در آینده میبینه و اینجا رو میکنه میگه من از دختر شاه سمنگان یک دونه پسر دارم یک کودک دارم پسر دارم و باشد اون کوده بچه هست الان هنوز آن گرامی نداند که چنگ توان باز کردن به هنگام جنگ اون کودک هرجمند الان اونقدری کچیکه که باید میدونم بدون اصلا به هنگام جنگ میشه دستارو از هم گشود و جنگید و کشتی گرفت و جنگید 

سوی مادر او به دست کسی فرستاده هم ذر رو گوهر بسید برای مادرش به دست کسی تلاه و جوائر فراوان هم فرستاده هم به خاطر این که از این بچه نگه داره میکنه 

و پاسخی که مادرش داده اینه که چون این پاسخ آمد که آن هرجمند بسی بر نیاید که گردد بلند طولی نمیکشه میگه از مادرش یه پاسخ اینچونینی و من رسید که زیاد طول نمیکشه که این بچه این کودک هرجمند ما قدر بکشه بزرگ بشه و جنگ آور بشه بنابراین تحمین رو نکرده که این بچه الان جنگ آور شده در دوزده سالگی گفته طولی نخواهد کشید اتمالا در آینده میتونه قدر به آسمان اتمالا سر به آسمان بساید 

همینه ای خورد با لب شیربوی شود بیگومان زود پرخاشجوی این پاسخیه که مادرش به من داده که با لبی که انوز بوی شیر میده شراب هم مینوشه با این که خورد ساله و از شیر خارگیش خیلی نگذشته شراب خوردن رو شروع کرده منظورش اینه بنابراین میگومان جنگ جو و پرخاشجو هم خواهد شد 

بنابراین مسئله نمیتونه اونقدی که کیک آووز شاه میگه یا گشده هم میگه جدی باشه نه تباهم زدن کسی از تورانیان نمیتونه به این رتبه برسته بباشیم یکی روزو دم برزنیم یکی بر لب خوشک نم برزنیم حالا بمون فعلا گیو تو زابول بمونیم امشب و بیاساییم به شراب لبی تر بکنیم این داره دوباره بزرگ نمایی میکنه معلوم نیست که این کابوس دوباره چه گندی زده که من باید برم جمعش بکنم 

بمونیم امشب و شرابی بنوشیم و از آن پس گراییم نزدیک شاه چی کارش کنه به گردان ایران نماییم راه بعد از این میریم به سمت شاه و سپایان ایران رو رهبری خواهیم کد کار جمع میشیم مگر بخت رخشنده بیدار نیست وگر نه چون این کار دوشوار نیست مگی این که این بخت شکومند من بر خلاف همیجه یار و همراه من نباشه وگر نه چون این جنگی نمیتونه برای من خیلی سخت باشه اینا دارم بزرگش میکنه چه دریا به موجندر آید زجای ندارد دم آتش تیز پای درفش مرا چون ببیند زدور دلش ماتم آرد به هنگام سور کینست چی هست این که میگید دریا با همواج سهمگین خودش به جنبش در بیاد نفس آتش شعلور و تاب و توان پایداری نخواهد داشت خاموش خواهد شد من رو ببینه درفش مرا چون ببیند زدور دلش ماتم آرد به هنگام سور از دور چشش به پرچه هم من بیافته هرکی باشه شور و شادی و اندوه براش بدل خواهد شد چون ماننده سام جنگی بود دلیر و حشیوار و سنگی بود به دین تیزی اندر نیاید به جنگ نباید گرفتن چون این کار تنگ میگه اگه اونجوری که گجده هم میگه و اینجوری که پادشاه میگه کیکابوسشاه میگه این مثل سام سوار باشه خیلی بسیار بیته مهمیه چون ماننده سام جنگی بود دلیر و حشیوار و سنگی بود به دین تیزی اندر نیاید به جنگ میگه اگه این چیزی که گجده هم میگه باشه واقعا یکی چی در حیبت سام پدر بزرگ من باشه دلاور و باوقار و خردمند باشه به دین تیزی اندر نیاید به جنگ قبلش فکر میکنه با این شتابزده ای که گجده هم میگه این یه همچین چیزی که آهنگ جنگ نمی کنه بنابراین باز داره تأکید میکنه این مدلی به جنگ اومدن که خردمندانه نیست که این باز هم تأییدیست براین داده که سهرا به نظر خردمند نمیرسه ولی جنگ آورت قدر قدرتیه نباید چون این کار نباید گرفتن چون این کار تنگ خیلی جدی نگیر این کارو سخت و دشوار نگیر اینجوری که اینا میگن 

نه خب میدونید که چرا این برداشت رو داره این برداشت رو داره چون طرف مقابلش کیکاووزشو اگه ایک شاه خردمند بود من احساس میکنم که رستم در کسری از ثانیه براه افتاده بود که همان که وقتی اسیر شده بود در جنگ مازندران و زال پدرش ازش خواست که همین الان بلند میشی میری دوتا مسیر داری یه مسیر بیشتر طول میکشه یه مسیر دو هفتهی میتونی بهش برسی ولی خب با چیزهای اجوغری بیرودر رو میشی که همون هفت خانه رستمه به زال که به پسرش رستم گفت رستم همون لحظه مسیر سریع تر رو با تمام ناهنباری هاش انتخاب کرد و بلا فاصله شروع کرد راهی شد نخورد و نیاسود رفت فقط برسه بنابراین این نشون میگه که رستم اصلا دل خوشی نه تنها از کیکابوسشاه نداره واقعا کیکابوسشاه رو در قدقامت یک پادشاه نمیدونه که فرمان او رو بلا فاصله اجرا بکنه چون میدونه که عمل کردش نابخردان است بنابراین ممکنه پیامش هم نابخردانه باشه و چیزی رو علکی بزرگ کرده و جدواره هدفش جنگیدن با جای دیگه است داره اینو میگه گشدم ترسیده که کابوسشاه از من چیزی میخواد این آدم آقلی نیست 

برای مانمیگه کارو سخت نگیه بیا تازه از را رسیدی بشین یه شراوی بزنیم به می دست بردند و مستان شدند زیاد سپه بود به دستان شدند به می گساری پرداختند شدت هم مست شدند پس از نوشیدن به سلامتی پادشاه به شادی زال پدر رستم هم نوشیدند زیاد سپه بود به دستان شدند دستان که می دینیم دوباره پسر رستم است 

شب گذروندن رستم گفتند شب بگذرون فردم دگر روز شبگیر هم پرخمار بیامد تهمتن برا راست کار سهرگاه روز بعد هم رستم اومد مست اومد و مدهوش اومد بسات دوباره بادنوشی رو فراهم کرد گسترد شروعیت به شادخاری کردند زمستی همان روز بازی استاد سوام روز رفتن نیامد شیاد اون روز به خاطر مستی جاموند نوشیده بود حرکت نکرد روز سوام که اصلا به یاد رفتن نیفتاد اینقدر حساب نمی کرد که کابوس شاه روز سه دیگر سهرگاه بیاورد می نیامد کابوس کی روز سوام که رفتن رو یادش نمی مد اصلا سهرگاه روز سه روز سوام روستم شراب رو آورد و به نهوی نوشید که اصلا یادش نمی مد که کابوس کیه بابا گلش کن 

ممکن براتون سآل بشه که این چه رفتار عجیبی از جهان پهلوان ولی خب باید باز مرور بکنید کارهای عجیب و غریبی که کابوس شاه کرده بود خشم عجیبی روستم نسبت به کابوس شاه داره و تا الان هم صرفا فقط احترام تخت پادشایی رو نگه داشته که با هاش رو در رو نشده و بهش چیزی نگفته هر بار روستم به درد سر افتده لشکر ایرانیان رو به کشت کشتار داده کیه کابوس شاه و خب روستم به تنهایی مجروع شده بره و کیه کابوس شاه و لشکر باقی مانده لشکر ایرانیان رو نجات بده 

روز چهاران شد حالا گیو هم بدبخت نگران که بابا بیا بریم کار پیچ خورده بعد اینم دیوونه میبونه است کیه کابوس شاه یه و یه کاری میکنه بیا بریم داستان درست نکنه به روز چهاران براراست گیو چون این گفت با گرد سالار نیو روز چهاران دیگه گیو بلند شد به روستم سرور دلاوران پهلوان اینجوری گفت به اون جسارتا جنابای روستم که کابوس تندست و حشیار نیست همین داستان بر دلش خار نیست قمی بود از این کار و دل پرشتاب شده دور از او خورد و آرام و خواه رو میکنه به روستم میگه ببین این کابوس آدم تند خویه حشیارم نیست خردمندم نیست همین داستان بر دلش خار نیست و این موضوع که فرمانش رو به تخیر انداختیم یا نه اساسا این مسئلهی که به وجود اومده براش خیلی بی اهمیت و کچیک نیست به نظر میسته به شدت نگران قمی بود از این کار و دل پرشتاب خش میگیره به من بیا بلند شد بریم شده دور از او خورد و آرام و خواه از این پیش آمد اونقدری اندوگین و نگران بود که از خواب و خورا گفته بود آرام و قرار نداشت به زابلستان گرد درنگ آوریم زمین پیش کابوس تنگ آوریم اگه بیشتر از این تو زابلستان بمونیم روزگ در کابوس تنگ بود و شوار بکنیم این خیلی نگران بودا رستم هم که جدی نمیگیره کابوس 

بلی نگرانی گیو رو که میبینه شما دقیق بکنید که چگونه چه تغییر و تغییراتی در ذهن رستم نسبت به که کابوس شاه رو اختاده یایتون هست اون باری که رفت به نخجیر در سرزمین توران؟ تا شادخاری میکرد تا شمه ای مینوشید یکی از پهلوانان بهش گفت ما که اینجا نخجیر و شکار رفتیم بیا این دفعه برای شکار بریم به سرزمین توران نه نگفت به پهلوان زیر دستش نه نگفت همون موقع اصبار رو زین کردن و رفتن الان هم که گیو میاد بهش میگه جون مادرت بیا بریم این ممکنه 

عرصه بهش تنگ بیاد ممکنه به من خشم بگیره بلافاس لسپ رو زین میکنه و میره یعنی وقتی گیوهش اصرار میکنه میره ولی درخواست پادشاه انقدر برایش کم عرج قرب شده به خاطر اینکه پادشاه خیردمند نیست اینا کم کم باید به شما داده بده از شخصیت رسلم که رسلم چگونه شخصیتیست ایران برایش مهم نگهبان ایران زمینه ولی سرسه پرده شاه نیست شاه رو اگر لازم باشه نقد میکنه حتی مانگل شاه رو جدی هم نگیره 

بدو گفت رستم که من دیش از این نگران نماشو چونه که با ما نشورد کسان در زمین کسی با من در نمیفت با هم بریم مسئله پیش نمیاد ولی نگرانی باش بفرمون تا رخش را زین کنند دمن در دم نای روگین کنند رستم دست رو داد روی رخش زین رو ببندن و به نشانه زمان حرکت در این نای های رومین بدمن اعلام بکنند که آقا جمع رفتیم سواران زابول یا زابول در نسخ دیگری ممکنه به گناه های دیگری دیده باشید شنیدند نای صدای نای به گوششون رسید برفتند با ترگ و جوشن زجای دیگه برحال دلاوران زابولستان سپاهیان زابولستان هم صدای نای رو شنیدند متوجه شدند که وقت رفتن جامعه جنگ رو در بر کردند در تند کردند را افتادند 

میرسند به پادشاه ولی کیکاوز شاه کار عجیبی میکنه گرازان به درگاه شاه آمدند گوشاد دل و نیکخاه آمدند 

چو رفتند و بردند پیشش نماز براشفت و پاسخ ندادی ایچ باز؟ روستم و گیو خرامان خودشون رو خوش خورم رسوندن به پادشاه و چهره و رفتارشون حاکی از این بود که دوستار و دلسوز شاه هستن در انجام دستورش که کتایی نکردن خودشون رو رسوندن وارد کاخ شاه شدن چو رفتند و بردند پیشش نماز رفتند ادای احترام کردند سجده هم کردند براشفت و پاسخ ندادی ایچ باز عجیبه وقتی پیشش رفتند ادای احترام کردند کاووس به خشم اومد هیچ پاسخی به ستایش و تعظیم اونها هم نداد اه رفتار عجیبه یکی بانگ برزد بگی و از نخست پسانگاه شرم از دو دیده بشست که رستم که باشد که فرمان من کند پستو پیچتز پیمان من بگیرش ببر زنده بردار کن وزونیز با من مگردان سخون خیلی عجیبه 

بلا فاصله که اینها رو دید اول سر گیوی فریاد بلند کشید و بعد شرمه هیارو گذاشت کنار که رستم که باشد که فرمان من رستم خرکی باشه که من بگم پاشو بیا تخیر بندازه چار روز بعد بیای کند پستو پیچتز پیمان من از پیمان من از عهد و پیمان من از دستوری که من دادم بخواد شانه خالی بکنه بگیرش ببر زنده بردار کن وزونیز با من مگردان سخون برو به گیو داره میگه بیا برو این رستم رو بردار ببر زنده زنده به دار بکش راجب رستم هم دیگه با من عرف نمیزنی یا خیلی عجیبه حالا اصلا گیرم که گیو بخواد این کار رو بکن مگه میتونه اون رو زنده زنده بردار بکشه زگفتار او گیو دل را بخست گیو هم شکه شد هم ناراحت شد چی داری میگی مرد من فرستادی برم این رو بگارم که بجنگه من برم رستم و زنده زنده دار بزنم اولا که اصلا چی داری میگی دوم من این که اصلا از این که بخواهم هم نمید مگه وای میسید من نگاه میکنه چی داری میگی مرد زگفتار او گیو دل را بخست که بردی به رستم برانگونه دست گیو ناراحت هم شد که بابا تو اصلا اگه سر تخت شاهی نشستی اگه رستم نبود تو الان سه بار مرده بودی مرد حسابی چی داری میگی جسارت رو از حد گذراندی به رستم همچنین چیزی میگی 

انجام نداد گیو همون جوری سرشون ناخته پایین هیچ کاری نکرد ول نمیکرد که ای کابوس برا شفت با گیو و با پیلتن در او خیره مانده همه انجامن که ای کابوس واکنش گیو رو دید که داره جانب داری میکنه از رستم من دارم میگم بیا بار اینو دار بزن نمیکنی غلط کردی هم به گیو خشم گرف هم به رستم این دفعه همه انجامن مونده بودن که این چیگاه داره میکنه یکی دیگر صدا کرد ول میکرد مگه بفرمود پس توس را شهریار که رو هرده را زنده بر کن به دار پادشاه رو کرد به توس توس و صدا کرد گفت بیا این اینو دار نمیزنه بلنش تو بیا باره جفتشونو دار بزن 

خود از جای برخواست کابوس که برفرخت برسان آتش زنی که کابوس از جاش بلند شد چهرشم از شدت خشم مثل آتشی که از نیزار داشت شوله میکشید سرخ و گلگون شده بود گر گرفته بود توس و صدا کرد گفت جفت اینا رو دار بزن بلنش رو رفت 

توس بدبختن چی بگه؟ اهمی الان دید یکی دستور دار زدن رستم رو داد به یکی دیگه اون نگو خودش هم الان باید دار توس بلند شد بره دیگه چه خاکی من تو سرم بریزم بشد توس و دست تهمتن توس بلند شد رفت دست رستم رو گرفت به نشانه این که خب الان بیا بریم بیرون دیگه 

بدون منده پرخاش جویان شگفت 

کسی انتظار نداشت توسم بره توسم از این شرایط ترسیده بود بلند شد رفت دست رستم رو گرفت که برداره ببره بیرون حالا زنده زنده یا بردار بکنه واقعا یا حالا فیلن ببره بیرون از این محلکه یه ذره این آتیش بخوابه ببینید چی که برداره بکنه ولی همین که توس بلند شد رفت دست رستم رو گرفت به نشانه اطاعت از فرمان پادشو همه در شگفت مونده بودن این نشان میده که رستم عجب جایگاهی داره که اصلا کسی تصور این رو هم نمی کنه اگر پادشو ها بخواد دستی دراز بکنه به رستم کسی انجامش بده بنابراین همه از کار توس هم در شگفت مونده بودن که چقدر لطی داری میکنی چرا بلند شدی بگی بشین سر جاد همه اونو که دار نمیزنه داره با رستم حرف میزنه آهر کسی نیست که 

توسه بدبخت هم رفته بود دست رستم رو گرفته بود بریم بریم چگاه باید بکنه که از پیش کاووس بیرون برد مگر کندران تیزی افسون برد اونم که میدونست نمیتونه بره رستم رو به دار بکشه که رستم رو از دربار کاخ اومد دبره بیرون از دربار کاووس بیمونده این که حالا این مجالی پیدا بکنه خشمتندیی که فضای کاخ رو پر کرده بوده براش چارجویی بکنه این کاری بکنه 

اینجا دیگه رستم قاطی کرد شما تصویر رو یه لذه تو زهنتون مجسم بکنین که شاه دستور داده اینا رو با هم دار بزنید بلند شده داره میره رستم قاطی میکنه تهمتن برا شفت با شهریار که تو دهن من رو سرویس کردی برد که خستم کردی که چندین من دار آتشن در کنار همه کارت از یکی دگر بدتر از تو را شهریاری نهندر خرستوزی مال شهریاری نیستی این تخت به تو زیادیه خیرتشو نداری رو این تخت بشینی هر کارت از اون یکی کارت بدتره برازنده و سزاوار مقام پادشایی نبودی سه بارم من تونه نجات دادم 

که چندین من دار آتشن در کنار یعنی هول ورد نداره دور ورد نداره نصد شما مال این عرفان نیستی

نامه فرستادی به من که من بلندشم بیام بعد میخوای من رو به دار بکشی که بره بجنگ با سهراب جنابای دکتر اگه میتونی سهراب رو زنده به دار بیاویز بعد برو دشمنت رو فرمانبر و رام خودت بکن توسم دستشو گرفته قاطی برا توسم میکنه بزد تند یک دست بر دست توست و گفتی زپیلی جیان یافت کوست با یه دست یه جوری به دست توست ضربه زد که انگار این ضربه را توست از یک فیل خشناک و سمگین خورد زبالا نگونندر آمد به سر برو کرد رستن به تندی گذر دستشو گرفته بود یه دونه زد به دست توست و جوری زد به دست توست همون جوری که ایستاده بود از ضربه رستن سرنگون شد با سر افتاد تو خاک و رستن با خشم از کنارشو از روش رد شد نه که پا روش بذارا از روش رد شد به مال این حرفا نسی چی میگه بدر شد به خشمندر آمد به رخش پاسخه که داره میده اصلا بسیار جدیه این عبیات خیلی جذابه بدر شد به خشمندر آمد به رخش یعنی رستن با خشم از درگاه کابوس بیرون رفت نشست روی رخش و چی گفت منم گفت شیروژن تاج بخش یادت نره تاجو که گذاشته رو سر تو یا تاجو کی رو سرت نگه داشته ها سه بار من از مرگ نجاتت دادم که الان تاج رو سر کلت ها به من میگن شیروژن تاج بخش تاج رو من میبخشم به سفر پادشاه من نباشم میگه میتونستی سر جاد بمونید حالا 

چه خشم آورد شاهکاووز کیست چرا دست یازد به من توسکیست اصلا کی هستین شما که این کارا رو دارید میکنید حواستون نیست میست که با کی طرفید شاهکاووز کیه که به من خشم بگیره توسکیه که به من گستاخی بکنه دست من رو بگیره به خود از درگاه ببره بیرون من رو ادام بکنه به دار بیاویزه زمین بنده و رخشگاه من است نگین گرز و مغفر کلاه من است بده کارتونم شدم من زمین فرمانبره و رخش تخت پادشاهی من من اگه میخواستم تخت پادشاهی برای خودم دست و پا بکنم که کرده بودم که 

تخت پادشاهی من رخشه من کارم جنگیدنه گستاخی به من نکنید نگین گرز و مغفر کلاه من است من تاج سرم نمیزارم من کلاه خود سرم میزارم اگه چشم به تاج و تخت کسی داشتم مگه من خیانت کردم که ببرید این وردارید اعدام بکنید زنده زنده بردار بکنید شب تیره از تیغ رخشان کنم به آوردگه بر سر افشان کنم شب تاریک و از برق شمشیر برانم روشن میکنم در میدون جنگ سر دشمن رو از تن جدا میکنم سر نیزه و تیغ یار منند دوبازو و دل شهری یار منند نتو 

سر نیزه و شمشیر هم یار و یار منند زور بازو و دلاوری و شجاعت من فرمان روای منه نه یه بیخردی مثل تو که آزاد زادم نه من بنده هم یکی بنده آفری ننده هم من بنده کسی آدم کسی نیستم و من آزاد زاییده شدم بنده پادشان نیستم که حق ندارد به من پرخاش و تندیو کند من فقط بنده آفریدگارم و بست تا الان هم هرچی از دستم برماده برای ایران زمین کردم به ایران نرید اون که سهراب گرد بیاید نماند بزرگ و نخورد شما هر کسی چاره جان کنید خرد را بدین کار پیچان کنید به من چه اصلا دهن من رو سرویس کردی هر بار یه تصمیم میگیری اگه سهراب دلاور این چیزی که برای من تو این نام نوشتید بیاد ایران بزرگ و کجی که تون رو زنده نخواهد گذاشت هر کدومتون برید برای حفظ جونتون چاره اندشی کنید برای دفت و رفع این مشکل خردتون رو خوشما به زحمت بننزید هرچی میشه میادید سراغه من بعد به من بیعترامی هم میکنید کم کردم مگه من به ایران نبینید از این پس مرا من میرم امون سیستان شما را 

شما تو زمید من تو آسمان جنو بسم الله این منبت با هم دیگه دستتون دیگه من نبیرسه خداحافظ شما 

خیلی سنگینه این وضعیت اصلا خیلی وضعیت پیچیدهیه پادشاه تحت فشاره پادشاه بیخرده رفتار نابخردانه میکنه دائمان ایران رو در معرض خطر قرار میده الان هم که توی همچی وضعیتی باید فرمان روای جنگ باشه بزرگ جهان پهلوانشو گرفته میگه تو دو روز دیر اومدی بیا با اینو دار بزنید خب چرا دو روز دیر اومده سه روز دیر اومده آدم حسابت نمیکنه چرا آدم حسابت نمیکنه به خاطر گندهی که تو الان زدی دیگه مرد خب بقیه که گناه نکردم بقیه هم مونده بودن که خب دیگه شروعیت خطیر شده حالا فلانی حمله کرده میگن جنگاور و تنومنده گشده همم دروغ نمیکنه ولی دیگه هده قما دلمون به رستم خوش بود اینم الان بره چه خاکی به سر بریزیم قمی شد دل نامداران همه که رستم شبان بود و ایشان رمه همه قمگین شده چون رستم حساب میکنه یهوب احساس بیپناهی کردن روز خواهی میکنه 

چی که کنیم؟ به گودرز گفتند که این کار توست آجان شکسته به دست تو گردد درست بیابرو اینم الان عصبانی رستم و احالین هرچی باشه پسر دستانه پسر زاله زال نماد خرده اینجوری نیست که این پهلوان خردمندی رستم باید بری از ذر خرد باش صحبت بکنی یه کاریش بکن این به دست توست بیابرو جمعش کن به گودرز گفتم که این کار توست شکسته به دست تو گردد درست این وضعیت به دست تو ممکنه فقط درست بشه سپه با جز از تو سخن نشنود همین بخت تو زین سخن نغنود اول بی برو بیش پادشاه بگو چه غلطی داری میکنی بابا پادشاه به جز تو از کسی حرف شنوی نداره در متقاعد کردن پادشاه امیدوارم که خوش اقبال و خوشبخت باشی به نزدیک این شاه دیوانه رو وزین در سخن یاد کن نو به نو بیا برو به پادشاه هیچیزی بگو باش فراوان سخن بگو سخنهای چرب و دراز آوری مگر بخت گمگشته باز آوری بیا برو سخنان نرم و دلنشین و تونیم مخین رو بزن از این تصمیم احمقانهی که گرفته گوتا بیاد این مسئله رو جمع بکنیم یعنی چی رستم رو بردار بکنیم یعنی نجنگه کدومه ما میخواییم به جنگیم درج سپید رو تخلیه کردن این یارو میرسه اینجا با خاک یکسانمون میکنه یکی باید وایس نجا داشت دیگه 

سپهدار گودرز کشواد رفت به نزدیک خسرو خرامید تفت گودرز کشوادگان شتابان رفت سمت کابوسشا به کابوسکی گفت رستم چه کرد که از ایران براوردیم روزگرد چرا اینجوری میکنی مرد تو؟ رستم چی کار کرده که به خشم اومدی اینجوری دمار از روزگار ایران داری در میاری و ایرانمون داری میکنی چه او رفت و آمد سپاهی بزرگ ابا پهلوانی به کردار گرگ که داری که با او به دشته نور چود برفشاند بر او تیر گرد خب اگه این بلنشه بره بعد از اون ور یه پهلوانی از تورانیان مثل گرگ بیاد به خادمارو بدره که داری که با او به داری که بره در دشته نورد باش به جنگه شود برفشاند بر او تیر گرد کی رو داری که بره اونو شکست بده؟ همین یه دونه رو داری اینم داری میگی زنده زنده جلو چشم دارش بخری تو یلانه تو را سر به سر گشده هم شنی دستو دیده است از بیش و کم 

که با او سواری کند رز میاد همون بیتیه که گفتم بعدن باش کار داریم میگه گشده هم اگه به شما همچین نامه اید ببین تو چقدر بیعتباری که وقتی تو برستم نامه میدی رستم تعلیل میکنه پهلوان تو برستم بگه من نگران جون خودم هم سریع ولن میشه میاد پدرش زال برستم بگه همین الان برو راه سختر انتخاب میکنه میاد تو بهش دستویر میدی چهار روز تعلیل میکنه چرا؟ 

اگه گشده همین نامه رو داده خب تو خودت اینو نمیفهمی که الان با جمعان پهلوانت نوادین که تو سرمایه اجتماعی نداری جمعان پهلوانت هم دیگه رو تو حساب نمیکنه که دستو رو تو بلافاصله اطاعت بکنه یلال تو را سر به سر گشده هم شنیده است و دیده است از بیش و کم نامه رو گشده هم برای تو فرستده گشده هم کیه؟ یلان تو و پهلوانان ایران زمین رو سن و سالی داره گشده هم هرچی داری و نداری دیده غربون شکلت برم با این که دیده اون بیت و اون دومصره رو گفته همی گویدان روز هرگز مواد که با اون سواری کنن رز میاد پهلوانان تو رو دیده و با احتساب این و با این دانش می روزی نیاد که این دلاوره تورانی یک پهلوان ایرانی رو در میدان جنگ پیدا بکنه چون فناش میکنه ها دیده و اینو گفته ها بنابراین فقط یه نفر داری ها اونم رستم و اونم از خود رنجون دیده 

کسی را که جنگی چون رستم بود بیازارد اون را خیرت کم بود انصافا منم دارم به خیرت تو این گودرز داره بیم که منم دارم به عقل تو شکل میکنم وجدانن عقل نداری شاهنشاهی که پهلوانی مثل رستم رو در حکمانیش داشته باشه و از خودش برنجوندش عقل در کله نداره آقا جان 

طبق معمول که 

خروشان میشه کیکاووس مثل من کردن شدم و از خواهیم گوشتون رو فکر کنم کر کردم انقدر دازدنم وقتی آشفته میشه تصمیم میگیره بلافاظران رو تصمیمش بر میگرده دیگه چو بشنید گفتار گودرز شا بدانست کودارد آین را پشیمان بشد زان کجا گفته بود به بیهودگی مغزش آشفته بود گفته های گودرز رو که شنید سخنانش رو معقول پسندید سخنانش رو و به این نجی رسید که نگو تو داری راست میگید من دوباره کمی دارم احمقانه رفتار میکنم پشیمان بشد زان کجا گفته بود به بیهودگی مغزش آشفته بود ترسیده بود مقتدرم نبود پاچایی بود که کاریزما هم نداشت رو بیدونید دیگه برحال فرمان روایی که کاریزما تیگ نباشه سرمایه اجتماعی هم نداشته باشه دیگه کم تجربه نکردید به این نقطه که میرسه معمولا هماغت میکنه تصمیمات غلطی میگیره پشیمان بشد دیگه غلطی پشیمان شد به بیهودگی مغزش آشفته بود از سر خشم بیخردانه رفتار کرده بود پشیمان شد بود جمع کنید مثلا دیگه به گودرز گفتی این سخن درخر است یعنی درخر است خر میخونم به واسطه همابایی با نیکوتر لبه پیر با پند نیکوتر است به گودرز گفتی این سخنها این نانی که گفتید به جا سزا باره برا زنده تره زبان پیر هم باره پندگو باشه خردمند باید دل پادشا که تیزی و تندی نیارد بهایه الان یاد اومد خسته نباشی دیر شدی که آقا پادشا باید خردمند باشه سنجیده رفتار کنه درست داره میگی خشم و تند خویی براش عرضش و اعتبار به بار نمیاره باشید برید بگید برگرده دیگه کارش بکنه شما را بباید پس او شدن به خوبی بسی داستان ها زدن سرش کردن از تیزی من تویی نمودن به او روزگار بهی برید دنبالش با نرمی و لطافت باشه صحبت بکنید سخنهای دلنشین و مجاب کننده بهش بگید برگرده بیاد دیگه دلشو از خشم و عصبانیت من پاکی بکنید دوران خوشی و خوبی آینده رو پیش چشمش مجسم بسازید که اگر این جنگ رو ببریم چه میشود چه نمیشود بلا برش دارید بیاریدش چهاره ندرم 

چا گودرز برخواست از پیش اوی پس پهلوان تیز بنهاد روی گودرز از پیشگاه شاه بلند شد و شاه از نظرش برگشته بود که خب باید بایدی اشتباه کردن یال برید برش دارید بیاریدش نمیخواد به دار بیا بیاریدش کسی از هم میتونه اینگر درنگ نکرد گودرز سریع تا اینو شنید بلند شد رفت زراغ رستم که بگید آقا برگرد پدرمون در میاد برفتن با او سران سپاه پس رستمندر گرفتن را سران سپاه هم همه ترس و لرز کنن که وای رستم بدبخ شدیم رستم رفت را افتادن دنبال گودرز برن برسن به رستم بگلت کردن بیافتن که آقا برگرد بیا این کار جمع بکنی 

چا دیدند گرد گوه پیلتن همه نامداران شدند انجومن رستم را افتاده بود بره واقعا جدی جدی گرد گوه پیلتن هم به هوا بلند شده بود داشت با رخش میرفت تاخت کنان رفتن در پیلت رستم گرد و گوهار هم به هوا برخواسته بود همه نامداران شدند انجومن همه بزرگان سپاه جمع شدند دور رستم که اینا یک کاریش بکنن ستایش که رفتن بر پهلوان که جاوید بادی و روشن روان آقا سب کن نر رسیدن شروع کردن به آفرین گوی و ستایش کردن از رستم که آقا نکن این کارو دیگه جهان سر به سر زیر پای تو باد همیشه سر تخت جای تو باد تو دانی که کاووس را مغز نیست به تیزی سخن گفتنش نغز نیست سراسر از جهان زیر دست و فرمان بردار و فرمان برد تو باشن الهی تو دانی که کاووس را مغز نیست تو میدونی کاووس عقل و خیلت درست داری مغز و کلش نیست که سخن گفتنش هم هنگام خشم این از سمیم دل نیست شایسته هم نیست این کار نیستی نستد ولش کن الان دق دقیق ایران باید داشته باشیم به جوشد همانگه پشیمان شده جوشیه به خوبی ز سر باز پیمان شده یه دقیقه خشم میاد یه سمیگه دو دقیقه باید پشیمان میشه یه بعد دوباره به راعدی انگار نه انگار برمیگرده به همون حرف قبلیش تهمتنگر آزورده گردد زشا باشه اصلا گیرم که این شاه اشتباه کرد آقا شما آزورده خاطرم شدی تهمتنگر آزورده گردد زشا هم ایرانیان را نباشد گناه گناه ما چیه میمیریم رستم بگی بری تو 

پشیمانم شده انواعی هیچی نشده رفتی بیرون هموزان سخنها پشیمان شده اصلا تندی بخواید همین پشت دست دست خوایدن پشت دست گاز گرفتنه یه حرکتی به نشانه ندامته و پشیمانی 

چاییم دمون شد 

رستم کتا نمیاد خیلی بهش برخورده که تو تو بد وضعیتی تو در بند دیو سپید بودی من هفتخان رستم رد کردن اومدم تو را نجات دادم شاه هاماوران با بیخردی تمام برای جشن رفتی اونجا اسیرت کد بدون سپاهیان رفتی اومدم تو را نجات دادم مثل دیوونا سر تختت اقاب بستی بلنشی بری آسمون احوالت کواکب رو هم میزون بکنی پاچه ها آسمان هم بشی افتاد یه زمین اومدیم پیدات کردیم نجاتت دادیم 

بعد جلو چشم من به من میگی زنده زنده اینو به دار بکشید کتا نمیاد دستم تهمتن چون این پاس و خواهرد باز که هستم زکاووسکه بی نیاز اون به من نیاز داره من به اون هیچ نیازی ندارم پشیمانی و تعصفش هم برای من مهم نیست و به پشیمانی و تعصفش هم نیاز ندارم گوره به درش هرچی گفته مرا تخت زین باشد و تاج ترگ قبا جوشن و دلنهاده به مرگ من رو به مرگ تعدید میکنین برای من تخت شاهیم زین تاج پادشاهیم هم کلا خودمه من چشم قردتی کسی ندارم که من که دنبال قردت میگردم که نگران موقعیت هم باشم که آی فلانی من رو از درگی گاهیش روند چی میشه؟ گروه پدره زر قبای من تصنیم و پذیرای مرگم هستم باکی ندارم ازش چه کابوس پیشم چه یک مشت خاک چرا دارم از خشم او ترس و باک کابوس برا من با یه مشت خاک یکیه؟ حالا میخواد کابوس جلوی من باشه میخواد یه مشت خاک باشه چرا باید از خشم کابوس من ترس و عراز داشته باشم که الان دارید به من میگید حالا او اصابالی شد ببخشید 

برای مهم نیست او اینکه آدم حسابش میکنم من اینو 

سرم گشت سیر و دلم کرد بس جز از پاک یزان نترسم زکی هست من فقط از یزان بایک میترسم دیگه از رفتارش عقل من به سطوه اومده دیگه دیوونه سین مرد 

زگفتار چون سیر گشت انجامن چون این گفت با چون این گفت گودرز با پیلتن چقدر عبیات مهمیه دقیق بکنید بعدن پس از ماجره های رستم و سهراب من یه تحلیل جدی رو این عبیات میخوام داشته باشم برای ما خیلی مهمه ازش ساده نگذرید چون جاهایی که به شما کم کم داده میده از رستم که چه چیزایی رستم رو تحریک میکنه چه چیزایی برای رستم مهمه چه چیزایی مهم نیست مقام براش مهم نیست تاج و تخت براش مهم نیست این که کی پادشاه مهم نیست آدم پادشاه نیست برده پادشاه نیست ولی 

ولی حفظ مرزبون و وزیفه براش مهمه برای همینم جنگ آوره و دائم اعلام میکنه که تخت پادشاهی من رخشه تاجی هم که من بر سرم میذارم تاج پادشاهی من کلا خود جنگ همه چرا؟ چون در حفظ مرزهای ایران باید بکوشم برای مهم نیست کی بر تخت شاهی نشسته خیرتمند باشه دست به سینه در اختیار او خواهم بود ولی بی خیرت باشه نقدش هم خواهم کرد و از او نخواهم حراسید بنابراین میتونید بفهمید که چه چیزایی براش مهمه و این انجابت بسیار مهمه که چی او رو تحریک میکنه که برگرده ز گفتار چون سیر گشتن جمن چون این گفتگو درز با پیلتن وقتی گروه سپه سالاران از گفتگو بس با رستم خصوصی دیدن بازخ نمیده از گفتگو سیر شدن متوجه شدن گفتگو جواب این بر نخواهد گشت چون این گفتگو درز رو میکنه به رستم و ببینید چی بهش میکنه 

که شهر و دلیران لشگر گمان به دیگر سخنها برند این زمان گفتگو درز رو میکنه به رستم سیاست بازی میکنه برای رستم دقیق بکنید از اینجا میتونید متوجه بشید که رستم 

چه چیزهای رستم رو بلاشت مثل تریگر عمل میکنه رو میکنه گودرز بهش میگه 

این که تو الان داریم میری یه وقت مردم و سپاهیان فکرهای دیگه ای میکنن چی فکر میکنه که از این ترک ترسنده شد سرفراز همی رفت زینگون چندی براز از این حرفا میزنن اون وقت که این رستم همیشه سرفراز و سربلند مثلی که از این گرد تورانی از این جوان تورانی به ترس و حراس افتده 

همی رفت زینگون چندی براز از این حرفا در گوشی ممکنه بر زبانها جاری بشه و شده باشه که چونان که گشده اهم داداگهی همه بومو برکرد باید تویی چو رستم همین زوبه ترسد به جنگ مراوطورا نیست جای درنگ یعنی هم تحریک میکنه رستم رو به این که یه وقت فکر میکنن ترسیدی ها که اینجوری داریم میری یعنی یکی از عواملی که برای رستم مهمه عزت و آبرو و قرور رستم رو میذاره رو میزاره رو میزاره رو میزاره رو میزاره رو میزاره ببین این حرفا هم جاری شده و خواهد شد که چونان که گشده اهم داداگهی همه بومو برکرد باید تویی چو رستم همین زوبه ترسد به جنگ ما خرکی باشیم که بایسیم 

وقت میگن اونجوری که گشده هم از سهراب خبرداده سراسر ایران رو باید تخلیه کرد و اگه رستم از جنگ با این میترسه دیگه ما کی باشیم که وایسیم به جنگیم ایران از دست خواهد رفت 

از آشفتن شاه و پیگار اوی بدیدم به درگاه برگفتگوی بلا فاصله که این اتفاقات افتاد در کاخ شاه و شاه به خشم اومد و با ستیز جویی رفتار کرد 

بدیدم به درگاه برگفتگوی همین تو زدی بیرون من این حرفا رو داشتم میشنیدم ز سهرابه ترکی هست یک سرسخون مکن پشت بر شاه ایران مکن 

چنین هم اومده چونین پشت بر شاه ایران مکن 

وقتی رفتی بیرون از کاخ من اینا رو شنیدم این حرفا رو داشتن میزدن که اگه این ترسیده دیگه کسی نباید به جنگ اینم از ترسش رفت بنابراین یک سره دارن از سهرابه ترکی سخن میگن ز سهرابه ترکی هست یک سرسخون یا سخون چونین پشت بر شاه ایران مکن نکن این کار 

اشاره به این میکنن که پشت بر کیکاووس شاه نمیکنی پشت داری بر شاه ایران میکنی و تو نباشی به جنگی همه تخلیه خواهند کرد هیچ کس با سهراب نخواهد جنگید و سهراب ممکن تخت ایران رو تصرف بکنه از شاه ایران اینجوری روبر نگردون 

چونین بر شده نامتن در جهان به دین بازگشتن مگردان نهان نام و آوازه نیک و افتخارات با این همه رنج و نبرد و پیروزی در جهان به سختی به دست اومده با این روی گردوندن از پادشاه از بینرفتن ایران تباه نکن نامتو نامتو به ننگ تبدیل نکن روستنا 

و دیگر که تنگ اندر آمد سپاه مکن تیره بر خیره این تاج و گاه دوم این که اصلا به این دعینا رو بلکن اصلا به این دلیل برگرد که سپاه دشمن نزدیک شده بیهوده این تاج و تخت و به خاطر یک پادشاه بیخرت به ورطه نابودی و سقوط نکش تو نباشی سقوط خواهیم کرد 

خب این اینای چیزایی که برای رستم مهمه به رستم بر این داستانها بخوان تهمتن دران کار خیره بماند سخنها رو به رستم گفت 

رستم هم موضوع براش مهمتر شد یعنی از خشم شکم فاصله گرفت از ادعاوتش الان با کیک و آووزشا یه مقداری کتا اومده که هم تصویر بزرگتر رو نگاه کرد سر کرد از قرورش به واسط انتفاق که افتده چشم بوشی بکنه و اون مفهوم بزرگتر یعنی ایران رو حفظ کنه 

تو کی؟ ندارم میگم من ترسیدم برای این دارم میرم به من بیعترامی کرده که دارم میرم ترس چیه؟ بدو گفت اگر بیم دارد دلم نخواهم که باشد زتن بکسلم 

رو کرد به گودرز گفت که در دلم من ترس و حراس رخنه کرده باشه گور پدر اون دل میکنم از سینه میندزمش بیرون از این ننگ برگشت و آن دید را که برگردد آید به نزدیک شا باش خیلی مهمه که کسی حتی چوه این به قول این روزی ها بیفته که رستم ترسی رستم ترسی رستم میترسه دیل من اگه ترس و باک درش هست اصلا نباید تو این سینه باشه از این ننگی برگشت و آن دید را از این ننگی که نزدیک و دامنگیرش بشه به واسطه خشونت کیک آووز شا که نزدیک و پشت به شای ایران بکنه برگشت و آن دید را راه درست را این دید که برگرده به سمت پادشاه نزدیک ای که آووز شا بره مسیحش رو برگردون رو دواره برگشت چو از دور شد شاه برپای خواست دقیقه بکنید بسی پوزش اندر گذشته بخواست از دور که رستم برگشت پدیدار شد از اون دور سریع که آووز شا رو تختش بلند شد سر پای اصداد و از اون چیزی که میانشون گذشته بود از رستم از خواهی کرد که تندی مرا گوهرست و سرشت من چیگاه کنم نمیدونم دیدی اینه که یه کاری میکنن نمیتونن درستش بکنن داره من از خواهی میکنم باز همون کار رو میکنن در آخرش به این نقطه میرسن که با با من همینم دیگه ولکن رها کن من اینجوری هم جدی نگیر منو خودش داره میگه منو جدی نگیر که تندی مرا گوهرست و سرشت چنان روز شاید که یزدان بکشت والا دست خودم نیست هستم من گیجم چی کار کنم این مودلیم برای از خواهی گفت اصلا این تندخویی تو تیزات تو طبیعت منم از که حالا دیگه اینجوری ایزد ما رو آفریده بذار همینجوری رفتار کنیم به دل نگیر و از این ناسه گالیده بدخواه نو دلم گشت باریک چون ماه نو یه وقت فکر نکنید فقط فردوسی هماسه داره میگه ابیاتی میگه با این کیفیت عجیب و غریب و از این ناسه گالیده بدخواه نو از این دشمن تازه به دوران رسیده و پیشبینی نشده دلم گشت باریک چون ماه نو چقدر قشنگ دلم مثل ماه سر برش که خیلی باریکه یک حلال ماه دلم اینجوری به تنگ اومده داغونم والا غمگینم بدین چار جستن تو را خواستم چو دیر آمدی تنگی آراستم گیش دارم میزنم نمیدونم چه غلطی کنم اینا هم که نمیتونم با این به جنگن وره همین ازت خواستم که بیایی تو هم دیر اومدید یه و عصبی شدم دیگه 

رستم هم بزرگواری میکن 

ولی کاملا در عدبیات رستم این رو میبینید که حسابش نمی کنه ولی دیگه الان کیکابوس نمیبینه الان تخت پادشاهی ایران میبینه و پادشاه ایران رو که حالا به فراخور وضعیت یک بیخرد روش بدو گفت رستم که گیهان تو جلو بشینست کنون آمدم تا چه فرماندهی روانت زدانش مبادا تویی حرفش رو هم میسن یالا اومدم یالا هرچی فرمان بدی به جای میارم فقط مبادا که هرگز خیرت در جانتو کاستی بگیره کرد غم میگیره کابوس هم شاد شدی به دو گفت کابوس که امروز بزم گذینیم و فردا بسازیم رزم حالا که دیگه اومدی بذاریم شبا به جشن و سرور بپردازیم فردا دست به کار جنگ میشین بیاراست رامش گهی شاهوار شدی وان به کردار باق بحار از آواز عبریشم و تمام خواهد شد ولی اگر بخوایم یه تحلیلی هم بکنیم این ابیاترو من دیگه قط نمی کنم پشته هم بریم کار رو جمع جور بکنیم یه تحلیل روانشناختی و جامع شناختی و فلسفی اگر بخواییم بکنیم این ابیاترو به نکات جذابی گاه من و تخت فرمان روایی من روی عصبم رخشه برای این که به جنگم فقط جایگاهش رو در مقابل شاه تسبیت میخواد بکنه درون رستم وفاداری به ایران و مسئولیت جهان پهلوانی زنده است ولی بی احترامی آشکار شاه رو هم بر شما نشون میده که این بی احترامی قدرت نفقط سرمایه اجتماعش رو از بین میبره که حتی وفاداری وفادار ترین شخص رو هم به اون کشور از بین خواهد بود یا در خطر خواهد انداخت تا لبه مرز قهر و ترک خدمت خواهد کشند تعلیل فلسفی هم اگه بخواید بکنید مسئله اطاعت و نافرمانی مشروع رو میتونید ببینید چون شاه در فرهنگ هماسی نماد نظم و یک پاچگی کشوره ولی وقتی فرمانش بر خلاف خیرد و مسئلات عمومی باشه
