میگه باره هر یک از این صد اسب و صد اسدر این صد تا اسب و این صد تا قاطر هر کدومشون بارشون پارچه های عبریشمین شاهانه ساخته چین و روم و پارچه های عبریشمین پارتی بودن ببردن صد بدره دینار نیز زرنگ و زبوی و زهرگونه چیز بدره میشه کیسه ی زر کسه ی زر که میگیم ولی زر و سیم سیم هم یه نقره هم توش میرختن ولی بدره جنس این کیسه از جنس پوست بود بنابراین وقتی میگه بدره میشه کیسه ی زر و سیم این که از جنس پوست باشه خلاصه صد کیسه زر سرخ و از زیور و زینتالاد و هر کونه مال و اسباب که وجود داشت خلاصه به عنوان هدیه برای رستم گذاشت کنار زیاغوت جامی پر از مشک ناب زیاغوت جامی پر از مشک ناب زی پیروزه دیگر یکی پر گلاب یک پیاله یاغوتین از جنس یاغوت پر از مشک خالص و جامی پیروزهی لبالب از گلاب رو بخشید به رستم نبشته این بخشش خیلی مهمه که پیمان نامهی هم تقدیم رستم میکنه نبشته یکی نامهی بر حریر زمشک و ز انبر ز اود و عبیر چی چی هست توش؟ سپرده به سالار گیتی فروز به نوی همه کشور نیم روز چنان که از پس عهد کاووز شاه نباشد بران تخت کس را کلا این از همش مهمتر 

انبر که میدونید یک ماده خوشبو از روده و مدهی فکر میکنم هم روده هم مدهی ماهی انبر استفاده میکنن براش اودم یک درختیه که از تیره پروانداران اگر اشتبانه کنم اصل این درخت هم در هندوستان و هندوچین هست از سوختن چوبش یک بوی خوشی متساعد میشه سیاه رنگ هم هست عبیر هم گونه خوشبوی از سندلها و گلاب و مشکه که با هم قاطیم میکردن و با زعفران هم معمولا میامیختن بوی بسیار خوبی میداد خلاصه به همراهش یک نامهی هم که بر حریر سپید با مرکب سیاه و خوشبوی آمیخته با مشک و انبر و اود نوشته بود رو هم تحویلش میده سپرده به سالار گیتی فروز به نوی همه کشور نیمروز چونان کس پس عهد کاووز شا نباشد بران تخت را کس کلا یا تخت کس را کلا میگه بنابراین دوباره پادشاهی سرزمین سیستان رو به پهلوان مایه امید و روشنایی بخش جهان واگذار میکنه چون نبود دیگه مدتی هم نبود و خب پس از اون هم درگیر جنگ بود و اینجا به شما نشون میده که احتمالا نارامی هایی هم در کشور ممکنه بوده باشه و بنابراین روی یک عهد با جوهر سیاه عملا با مرکب سیاه و خوشبوی آمیخته به اود و عبیر و انبر یک فرمان مینویسه و این فرمان پادشاهی سیستان یا همون نیمروز که اینجا اشاره بش میکنه همه کشور نیمروز منظورش سیستان هست سرزمین سیستان هست یک عهدنامه می نویسه که منبر پادشاه سیستان رستم خواهد بود فرمان روای سیستان رستم خواهد بود چنان که از پس عهد کابوس شاه نباشد بران تخت کس را کلا یه جوری هم این عهدنامه رو می نویسه که داخلش این قید شده که پس از دوران کابوس شاه هم امکان پادشاهی سیستان رو هیچ کس به غیر از رستم نداشته باشه یعنی پادشاهی هیچ کس به غیر از رستم بر سیستان ممکن نباشه حتی اگر در حقیقت کابوس شاه از بین بره خب می دونیم که اگر کابوس شاه از بین بره پادشاه بعدی بیاد و خب جنگی برای قدرت روخ نداده باشه و پادشاه بعدی انتخاب بشه و پادشاهی ایران برزه اتفاقا همیچون موضوع اعتمالا روخ نخواهد داده نگر این که جن 

غیر از رستم مجددن بشه پادشاهی سیستان بنابراین پادشاهی دائمی سیستان رو می سپره به رستم 

چنان کس پس عهد کابوس شاه نباشد بران تخت کس را کلاه مگر نامور رستم زال را خداوند شمشیر و کوپال را خداوند اینجا می دونید بارها در شاهنامه صحبت کردیم در منعه مالک صاحب و دارنده استفاده شده و خب کنایز استاد تمام اینجا خداوند شمشیر و کوپال را بهترین به کاربرنده شمشیر و کوپال ماهر و استاده میگه به جز رستم نامدار پسر زال این استاد چیر دست در به کار بردن شمشیر و گرز کسی تا زمان که رستم زنده است حق نداره پادشاه سیستان بشه کلن سندش رو ششتانک می زنه به اسم رستم وزان پست برو آفرین کرد شاه که بی تو مبادا نگین و کلا دیگه چیزی بگید چیزی دیگری غیر از این نمیشه بگید دل تاجداران به تو گرم باد روانت پر از شرم و آزن باد دمت گم که تو اگه نبودی من اصلا زنده نبودم چه برسه روی تخت پادشاهی نشسته باشم نگین و کلا اینجا مهر و تاج هست مجازن پادشاهی شاه ستایشش کرد کیکاووز شاه رستم رو ستایش کرد بعد از این که این نامه رو بهش داد و بهش گفت پادشاهی ایران اصلا بدون وجود تو نباشه دل تاجداران به تو گرم باد روانت پر از شرم و آزرم باد گرم یعنی محکم و استوار قوی و آزرم هم یعنی شرم دل پادشاهان به اتقای تو قوی و استوار باشه جان تو همواره لبریز از شرم و نرمی و عدب باشه که همیشه پادشاه به تو نز قدرت تو استفاده بکنه یعنی نسبت به نهاد پادشاهی دلت نرم و پر شرم باشه 

فرو بود رستم ببوسید تخت فرو برد هم میبینید در برخی از نسخ در نسخه بریروت فرو بود هست چون اگه خاطرتون باشه وقتی رستم اومد گفت پایین تخت نشستی که پادشاه روی تخت پادشاهی بود رستم پایین بود فرو بود رستم ببوسید تخت به سیچه گذر کرد و بربست رخت کم صحبت هم هست کارشو میگه کارشو انجام میده میره به سیچه گذر کردن یعنی آماده رفتند شدن رستم همونطور که پایین تختگاه شاه نشسته بود میگه بر تخت پادشاهی بوسه زد و پس از اون آماده برگشتن به سیستان شد و راه افتاد خیلی کشش نداد خروش تبیره بر آمد ز شهر ز شادی به هر کس رسانید بر بر آمد هیاهوی و بانگ درای به گوشندرون ناله کرنای بانگ تبل از پای تخت بلند شد که خب رستم داره میره و هر کسی سهمی از سرور رسید هر کسی از شادی و سرور سهمی بر بر آمد هیاهوی و بانگ درای هیاهو و قوقا و نوای زنگ بلند شد و در گوش صدای شیپور پیچید و بشد رستم زال و بنشست شاه جهان کرد روشن به آین و راه بشد اینجا شدن یعنی رفتن دیگه 

رستم پسر زال از پای تخت فاصله گرفت شد راهی سیستان شد و شاه به اداره کشور پرداخت بریم ببینیم چگاه باید بکنیم کشور جهانداری بکنیم فرمان روایی بکنیم بلاخره نشستیم رو این تخت ببینیم چه خبره جهان رو با برقراری رسم و سنت و قانون آبادان کرد شروع کرد واقعا فرمان روایی دادگرانه کردن بزد گردن قم به شمشیر داد نیامد همین بردل از مرگیاد مردم حالشون خوب بود با شمشیر ادارت سر از تن اندوه ایرانیان جدا کرد بزد گردن قم به شمشیر داد یعنی ادارت پیشه کرد و قم رو از مردم دور کرد گردن قم رو زد همین نیامد همین بردل از مرگیاد به کلی تغییر روش داده بوده یه حواظش دیگه بشت سرمایه اجتماعیش بود اونقدر حواظش به سرمایه اجتماعیش بود که حواظش بود اصلا کسی از فرد خوشی و شادی مرگ رو به یاد نیارست زمین گشت پر سبزه و آب و نم بیاراست گیتی چو باق ارم توانگر شد از داد و از ایمنی زبد بسته شد دست آهرمنی جهان رو با آراستگی و آبادانی باق ارم کرد باق ارم اسم خاصین باق اساتیری یه معروفه میکنم الان هم در شیراز فکر میکنم که ندیگه در شیراز هم یک باق جذاب باقی بسیار زیبا به نام ارم کماکان داریم 

اصخایی میکنه خلاصه میگه جهان رو اون قلم روی فرمان روایش رو به آراستگی و آبادانی باق ارم کرد با کارهایی که کرد ادالتی که برقرار کرده بود زمین هم در همراهی با عدل شاهی که خب میدونید این دوتا با هم همیشه همراه هستن اگر خاطرتون باشه اون جنگ عظیم رو که مدتها زمان برده بود بین ایران و تورن و خوشکسالی شده بود به واسطه این جنگ و خونریزی چون عدل از زمین رخت بسته بود رخت بر بسته بود اینجا میگه زمین هم به واسطه همراهی با عدل پادشاهی از سبزه و آب و تازگی و تراوت آکنده شده بود توانگر شد از داد و از ایمنی زبد بسته شد دست آهرمنی مردم در نتیجه دادگری و امنیت سروتمند شده بودن دست احریمن هم از بد کرداری کتاه شده بود داشت مسیرو درست میرفت جلو بگیتی خبر شد که کاووز شاهز مازندران بستدان تاجوگاه بماندند یک سر همه زین شگفت تازه خبرش داره پخش میشه که ای چه خبر شد این ستدن یا استدن یا ستادن یعنی گرفتن دیگه زمازندران بستدان تاجوگاه میگه تازه خبرش رسید به سراسر جهان که کاووز شاه تاجو تخت مازندران رو هم تسخیر کرد و گمارده خودش رو اونجا نشوند خبر عجیب بود چون کسی تا اللہ این کار رو نکرده بود اصلا برای همین داشت میرفت این کار رو بکنه که این خبر روزی در جهان پخش بشه نام او به یادگار بمونه که بالاخره من رفتم هیچی نرفته بود جورت نکره بود من رفتم مازندران رو فت کردم و حال این خبر پخش میشه دیگه بماندند یک سر همه زین شگفت که کاوز شاهان بزرگی گرفت همه پاک با هدیه و با نسار کشیدند صف بردر شهریار جهان چون بهش دیش و داراسته پر از داد و آگنده از خواسته اتفاقی که میفته یعنی که خبر میرسه به جای جای جهان که ببین این مثل که یه فرقایی میکنه حالا نمیدونن دیگه که همه که بند جزئیات نیستن که خوبیه اسیری شده با رسته اومد کار درابور حالی برحال این به اسم پادشاه ایران به اسم کیکابوزشان نوشته میشه بنابراین همه خبردار شدن که کیکابوزشان رفت سرزمین مازندران رو هم فرد کرد و گمارده و گماشته خودشون رو همونجا نشوند و تحت ید و زیر ید قدرت پادشاه مازندران منبد پادشاه ایران مازندران هم منبد خب شگفزده شدن که این سرزمین دیوار چیجوری گرفت چیگار کرده اصلا چه اتفاق یختاد و این باعث شد که یه اعتبار عجیب و غریبی هم به دست بیاره یعنی اعتبارش بیش از پیش بشه و همه از این که این همچین کاری رو کرده و سرزمین مازندران رو تسخیر کرده همه چونان پایگاه عظمتی پیدا کرده غرق شگفتی شده بودن و بنابراین به این نجای رسیدن که بهتر بیان ارز ارادت و احترام میکنن پادشانه جهان فرمان روایان گوشه گوشه جهان همه بدون کم و کاس با هدیه و پیشکش اومدن در درگاه کاوزشاه جهان چون بهشتی شد آراسته همه صف کشیده بودن پشت در بهش هدیه و نصار میدادن و جهان چون بهشتی شد آراسته پر از داد و آگنده از خواسته جهان مثل یک بهشت زیبا و آبادان شد و از دادگریه که ای کاوزشاه و دارایی آکنده شده بود برحال سرمان اجتماعش رو به دست آورده بود همه وضعیت اقتصادیشون در ایران زمین خوب شده بود به واسطه دادگریه که پیشه کرده بود هم عدل و داد پیشه میکرد هم برحال درگنج رو گشوده بود به نیازمندان کمک کرده بود برحال الان در ایدئال ترین حالت ممکنش که ای کاوزشا بر تخت پادشایی نشسته و رسمن جنگ مازندران خاتمه پیدا میکنه و همینجا مهر پایان جنگ مازندران رو هم میزنه پای این داستان فردوسی سر آمد کنون رزم مازندران به پیش آورم جنگ هاماوران به بخش از خواهی میکنم سر آمد کنون رزم مازندران دیگه داستان رزم مازندران به پایان رسید حال وقتشه که یه چیزی رو شروع بکنم که بعد جوری عجیبه به پیش آورم جنگ هاماوران جنگ هاماوران رو من حال و باید آغاز بکنم که یکی از پیچیده ترین و جذابترین و عجیبترین بخش های شاهنام از کمیش رو میخونم بریم با هم جلو تا نزدیک صد بیت رو من در این قسمت خونده باشم و بقیهش رو فرد و شب ادامه میدیم چی میشه که اصلا این جنگ هاماوران روخ میده؟ 

مگه نه این که الان کهی کاووز شاه روی تخت پادشایش نشسته؟ مگه نه این که از گوشه گوشه جهان فرمان روایان دارن میان براش خواسته و حدیه هم میاره؟ چه اتفاقی میفته که همچین رزمی و همچین جنگی روخ میده؟ 

از آن پس چونین کرد کاووز را که بر تخت زرین به جنبت زجایی؟ مدتی تو همین وضعیت میمونه ایران هم سر و سامونش رو پیدا کرده و همه مردم تو شرایط خوبی هن، سبزی و آبادانی هم سر تا سر این قلم رو رو فرا گرفته؟ از جای جنبیدن این گروه فعلی یعنی جایی به جایی رفتن، حرکت کردن، سفر کردن، لشگر کشی کردن، سر تخت پادشاهی نبودن که بر تخت زرین به جنبت زجایی. بعد از این که کشور رو آباد و ایمن کرد، تصمیم گرفت که هم یه سری به قلم روی فرمان روایش بزنه، ببینه اوزا چیه؟ چون میدونیم نافرمانی هایی هم یه جاهایی روخ داده بود برحال وقتی که او پادشاه رفته بود به جنگ و مدتها نبود دیگه کشور رو کیداشه داره میکرد؟ ممکنه اونجا نزایی هم برای قدرت شکل بگیره دیگه.

در بیت های اشاره میکنه البته که میدونیم کل داستان شاهنامه به ناپایداری قدرت و عملا ناپایداری جهان اشاره میکنه ولی اون بیت جذابی که میگفت بر گوشه گل ستان خار روست و اون بیتی که گفتم یک گوشه آزمایش بکنید راجع به جواز نیافتن از آزمایش صحبت میکرد معید این دیدگاه هست که در اوج پیروزی هم بزر خطر و شکست میتونه در پس پرده نهفته باشه که این یاداوری چرخه فراز و نشیب عملا بخت و اقبال شاهنامه مرتب در تمام داستانهاش به مای موضوع رو گوش زد میکنه که جهان دائمان در گردشه و فرجام قرور میتونه سقوط باشه نه فقط برای ایک پادشاه ولی برای خود ما هم همین جوریه یعنی هر زمان که در پیروزی مطلق به سر میبریم احتمال شکست نمیدیم ولی شاهنامه یکی از مواهب شینه یکی از پندهایی که دائمان میده در هر پادشاهی میده همیشه اگر دقت بکنید پایان مسیر ایک پادشاه لحظه مرگ ایک پادشاه یه مرور جزئی به کارنامه اون پادشاه میکنه و میگه که چگونه جهان برش سپری شده اصلا آیا اصلا به چشمش میومده که این جهان سالیان درازی برش سپری شده چقدر خودش رو مهور آلم میدیده چقدر نمیدیده چقدر با مرگ را حد کنار میاد چقدر نمیاد و این خیلی به رفتاری که قبل از مرگ داشته و در این دوره فرمان روایش داشته ارتباط پیدا میکنه و این که اون پادشاه هم زمانی بر تخت فیروز نشانی میشسته ولی امروز کل صحبش از جهان یک گور خواهد بود 

بود تاریخی رو هم اگر بخواییم بررسی بکنیم دنوان بود آخری که ترجیح میدم دیگه بعد از اون تحلیل رو به اتمان برسونم بریم برای فرداشت که قصدتونم نکنم این گسترش قلم رو در روایت هماسی هست که ممکنه آزارتون هم داده باشه تو منابع تاریخی تصرف هایی که کیکابوزشابش دست میزنه ممکنه ریشه در واقعیت های استوره های فتوحات فرمان روایان باستانی داشته باشه شاهنامه هم این مناطق رو بر پای زهنیت ایرانی اون روزها جز به مرزهای بلقوه ایران برمیشه مورده این رو هم در نظر داشته باشید نام هایی مثل مکران و بربرستان و مصر و شام نشون میده که فردوسی دایره حوییت و اندیشه ایران فرهنگی رو داره وسیع میبینه دیگه ولو این که ممکنه در زمان فردوسی این قلم روها زیر فرمان ایرانیان نبوده باشه در نظر داشته باشید فردوسی زمانی دست به سرایش این عبیات زده که سلطان محمود قزنوی بر عریقه قدرت نشسته و سپاه اسلام 400 سال قبل ایران رو شکست داده و حواسمون هست که فردوسی در این مقتع نوشته اما به نظر میاد که فردوسی مرزهای بلقوه ایران رو بخیلی بزرگتر از اون چیزی که اون لحظه درش نشسته و داره زندگی میکنه و این عبیات رو مینویسه در نظر گرفته که این نام ها وراش میاد و خب راجبش این گونه صحبت میکنه بلاورین ممکنه که این قلم روها در زمان فردوسی در زمان زیستن فردوسی زیر قلم روی فرمان ایرانی ها نبوده باشه ولی اون حوویت و اندیشه ایران فرهنگی رو خیلی وسیع تر میبینه برای این که بتر درک بکنید مثل خود ما میبونه اونایی که امروز هم ایران شهری فکر میکنن تفکر ایران شهری دارن خب متقد هستن حال تفکر ایران شهری هم نداشته باشن با عنوان مثال وقتی به افغانستان نگاه میکنن به شرق ایران نگاه میکنیم میدونیم که خب تقریبا ست سال مرزای جدایی پیدا کردن بنابراین اگر شما با دیدگاه وسیع ترین نگاه بکنید میبینید که خب این نگاه ممکنه هم 

بکنه یعنی برخی که تفکر ایران شهری ممکنید داشته باشن خیلی سفت و سخت یا اصلا با این منظومه فکری در تماس باشن و منظومه فکری شدن پی اوویشون در ایچه نگاهشونین باشه حتی ممکنه مرزها رو یکی ندونن ولی ممکنه مردمانی که در افغانستان و عراق همزیست میکنن رو به نوعی هموطن قدیمی خودمون بشمارن بنابراین این رو دارم بیان میکنم که درک دقیقی از منظومه 

مکنان و این مناطقی که داره مثال میزنه که تحت ید عراب هست در اون لحظهی که فردوسی داره صحبت میکنه و سخن میگه برای ما ولی این نشون میده که ذهنیتش و ذهنیت ایرانی آن روزگار شاید اینگونه بوده که دایره و حوییت و اندیشه ایران فرهنگی خیلی وسیع تر از اون چیزی بوده که اون لحظه مرزا تعریف میکرده بازتاب اندیشه سیاسی ایران بازتان رو هم شما میبینی چون بازتاب دهنده آرمانیه که توش پادشاه مشروع وظیفه داشته امنیت رو در گسترهی فراخ برقرار کب بکنه و ناراضیان و گردنکشان رو سرکوب بکنه در این حال داد و دهش رو هم پیشه بکنه دل مردم رو هم به دست بیاره و در انتها موضوعی که اتفاقه الان درگیرش هم هستیم ایدن و روز از نظر فرهنگی جشن و سرور برای بازگشت پادشاه پیروز حضور رامشگر ساز و می همه نشونگر سنت درباری ایرانیه که تو مطنع که هم به شکلهای گناگونی توصیف شده کما این که میدونیم ما برای هر کدوم از ماهامون جشن داشتیم شبیه به الان نبود اینقدر درگیر ازا و ازاداری گذشتگان نبودیم همیشه عباد موفقیت و پیروزی بیشتر بیان می شود و براش جشن و سروری برپا می شود هر ماه این قاعده داشتیم در هر ماه یک جشن بزرگ داشتیم و این نشون می داد که سنت درباری ایرانی هم در گذشته این گونه بوده سنت شاهی و فرمان روایی ایرانی سنتی و کهان همین گونه بوده که به جشن و سرور اهمیت ویجه می دادن حتی اگر پادشاه از پیروزی باز می گشت هم هفته ها جشن و سرور برپا می کردن چه برسه به وقتی که جشن های بزرگ ایرانیان پاش در میان بود که سر می کردن حتی دقیقا پادشاهان سرمایه اجتماعیشون رو نگه دارن و مردم رو شاد نگه دارن مثل الان نبود که همیشه عزاداری بکنن بنابراین امیدوارم که اید نروز هم به شما همین حس رو بده بدونید که باید این سنت رو حفظ بکنید به واسطه این که بخشی از سمد دون ماست و به جشن بنشینید حتی اگر اوضاع اقتصادی سیاسی و اجتماعی نارامه و خیلی شاید انگیزه برای این اتفاق نباشه ولی با اسمش هم بستگیم رو بیشتر با هم حفظ بکنیم فردا شب می پردازیم به جنگ هاموبران شبه خوبی داشته باشید سال نرو مجددن خدمتون تبریک می گم
